inteiro fatorial(inteiro n) {
    caso |n <= 1| executa & {
        devolve 1;
    }
    outro {
        devolve n * fatorial(n - 1);
     }
}

inteiro principal() {
    ATRIBUICOES:
    inteiro notaprovamaurio := 15
    CODIGO:
    inteiro resultado;
    inteiro num := 5;
    resultado := fatorial(num);
    out resultado;
}
